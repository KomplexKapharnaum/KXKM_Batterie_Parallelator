{
    snprintf(buf, len, "N/%s/%s", portal_id, path);

    snprintf(buf, len, "{\"value\":%.2f}", val);

{
    float soc = (avg_v - v_min) / (v_max - v_min) * 100.0f;

    if (soc > 100) soc = 100;
    return soc;
}

/**
 * @file test_ble_victron.cpp
 * @brief Host tests for Victron BLE Instant Readout payload encoding.
 */
#ifdef NATIVE_TEST

#include "unity.h"
#include <cstdint>
#include <cstring>

#define VICTRON_COMPANY_ID      0x02E1
#define VICTRON_RECORD_SOLAR    0x01
#define VICTRON_RECORD_BATTERY  0x02

/* ── Battery Monitor payload packing (mirrors bmu_ble_victron.cpp) ── */

static int pack_battery_payload(uint8_t *plain, float avg_v, float total_i,
                                float soc, float ah_c, float ah_d)
{
    uint16_t remaining_ah = (uint16_t)(ah_c * 10.0f);
    uint16_t voltage = (uint16_t)(avg_v * 100.0f);
    int16_t current = (int16_t)(total_i * 10.0f);
    uint16_t soc_u16 = (uint16_t)(soc * 10.0f);
    uint16_t consumed_ah = (uint16_t)(ah_d * 10.0f);

    memcpy(plain + 0, &remaining_ah, 2);
    memcpy(plain + 2, &voltage, 2);
    memcpy(plain + 4, &current, 2);
    memcpy(plain + 6, &soc_u16, 2);
    memcpy(plain + 8, &consumed_ah, 2);
    return 10;
}

/* ── Solar Charger payload packing ──────────────────────────────── */

static int pack_solar_payload(uint8_t *plain, uint8_t cs, uint8_t err,
                              uint32_t yield_wh, uint16_t ppv,
                              float ibat, float vbat)
{
    plain[0] = cs;
    plain[1] = err;
    uint16_t yield = (uint16_t)(yield_wh / 10);
    int16_t ibat_i = (int16_t)(ibat * 10.0f);
    uint16_t vbat_u = (uint16_t)(vbat * 100.0f);
    memcpy(plain + 2, &yield, 2);
    memcpy(plain + 4, &ppv, 2);
    memcpy(plain + 6, &ibat_i, 2);
    memcpy(plain + 8, &vbat_u, 2);
    return 10;
}

/* ── Manufacturer data header ───────────────────────────────────── */

static int build_mfr_header(uint8_t *buf, uint8_t record_type, uint16_t counter)
{
    buf[0] = (uint8_t)(VICTRON_COMPANY_ID & 0xFF);
    buf[1] = (uint8_t)(VICTRON_COMPANY_ID >> 8);
    buf[2] = record_type;
    buf[3] = (uint8_t)(counter & 0xFF);
    buf[4] = (uint8_t)(counter >> 8);
    return 5;
}

/* ── Tests ────────────────────────────────────────────────────────── */

/**
 * @file test_config_labels.cpp
 * @brief Host tests for battery label config: bounds, truncation, file format.
 */
#ifdef NATIVE_TEST

#include "unity.h"
#include <cstring>
#include <cstdio>

#define BMU_MAX_BATTERIES       16
#define BMU_CONFIG_BATLABEL_MAX  9

/* ── Simulated label storage (mirrors bmu_config.cpp) ────────────── */

static char s_labels[BMU_MAX_BATTERIES][BMU_CONFIG_BATLABEL_MAX];

static void init_defaults(void)
{
    for (int i = 0; i < BMU_MAX_BATTERIES; i++) {
        snprintf(s_labels[i], BMU_CONFIG_BATLABEL_MAX, "B%d", i + 1);
    }
}

static int set_label(int idx, const char *label)
{
    if (idx < 0 || idx >= BMU_MAX_BATTERIES) return -1;
    if (label == NULL || label[0] == '\0') return -1;
    strncpy(s_labels[idx], label, BMU_CONFIG_BATLABEL_MAX - 1);
    s_labels[idx][BMU_CONFIG_BATLABEL_MAX - 1] = '\0';
    return 0;
}

static const char *get_label(int idx)
{
    if (idx < 0 || idx >= BMU_MAX_BATTERIES) return "?";
    return s_labels[idx];
}

/* Parse file content (one label per line) into label storage */
static int parse_labels_from_buffer(const char *buf)
{
    init_defaults();
    int idx = 0;
    const char *p = buf;
    while (idx < BMU_MAX_BATTERIES && *p != '\0') {
        const char *eol = strchr(p, '\n');
        if (eol == NULL) eol = p + strlen(p);
        int len = (int)(eol - p);
        /* Strip \r */
        if (len > 0 && *(eol - 1) == '\r') len--;
        if (len > 0) {
            if (len > BMU_CONFIG_BATLABEL_MAX - 1) len = BMU_CONFIG_BATLABEL_MAX - 1;
            memcpy(s_labels[idx], p, len);
            s_labels[idx][len] = '\0';
        }
        idx++;
        p = (*eol == '\n') ? eol + 1 : eol;
    }
    return idx;
}

/* ── Tests ────────────────────────────────────────────────────────── */

void test_defaults(void)
{
    init_defaults();
    TEST_ASSERT_EQUAL_STRING("B1", get_label(0));
    TEST_ASSERT_EQUAL_STRING("B16", get_label(15));
}

void test_set_valid(void)
{
    init_defaults();
    TEST_ASSERT_EQUAL_INT(0, set_label(0, "Tender"));
    TEST_ASSERT_EQUAL_STRING("Tender", get_label(0));
}

void test_set_truncation(void)
{
    init_defaults();
    set_label(0, "VeryLongBatteryName");
    TEST_ASSERT_EQUAL_INT(8, (int)strlen(get_label(0)));
}

void test_set_invalid_idx_negative(void)
{
    TEST_ASSERT_EQUAL_INT(-1, set_label(-1, "X"));
}

void test_set_invalid_idx_overflow(void)
{
    TEST_ASSERT_EQUAL_INT(-1, set_label(16, "X"));
}

void test_set_empty_label(void)
{
    TEST_ASSERT_EQUAL_INT(-1, set_label(0, ""));
}

void test_set_null_label(void)
{
    TEST_ASSERT_EQUAL_INT(-1, set_label(0, NULL));
}

void test_get_invalid_idx(void)
{
    TEST_ASSERT_EQUAL_STRING("?", get_label(-1));
    TEST_ASSERT_EQUAL_STRING("?", get_label(16));
}

void test_parse_file_basic(void)
{
    int n = parse_labels_from_buffer("Tender\nLED1\nLED2\nCabine\n");
    TEST_ASSERT_EQUAL_INT(4, n);
    TEST_ASSERT_EQUAL_STRING("Tender", get_label(0));
    TEST_ASSERT_EQUAL_STRING("LED1", get_label(1));
    TEST_ASSERT_EQUAL_STRING("LED2", get_label(2));
    TEST_ASSERT_EQUAL_STRING("Cabine", get_label(3));
    TEST_ASSERT_EQUAL_STRING("B5", get_label(4)); /* default kept */
}

{

    TEST_ASSERT_EQUAL_STRING("B2", get_label(1)); /* empty = keep default */

    TEST_ASSERT_EQUAL_STRING("LED1", get_label(1));

    TEST_ASSERT_EQUAL_INT(16, n); /* capped */
}

{

    RUN_TEST(test_set_truncation);

    RUN_TEST(test_set_invalid_idx_overflow);
    RUN_TEST(test_set_empty_label);
    RUN_TEST(test_set_null_label);
    RUN_TEST(test_get_invalid_idx);

    RUN_TEST(test_parse_file_empty_lines);

/**
 * @file test_vedirect_parser.cpp
 * @brief Host tests for VE.Direct TEXT protocol parsing + checksum.
 */
#ifdef NATIVE_TEST

#include "unity.h"
#include <cstring>
#include <cstdio>
#include <cstdlib>

/* ── Simulated parser state (mirrors bmu_vedirect.cpp) ───────────── */

typedef struct {
    float    battery_voltage_v;
    float    battery_current_a;
    float    panel_voltage_v;
    uint16_t panel_power_w;
    uint8_t  charge_state;
    uint8_t  error_code;
    uint32_t yield_today_wh;
    bool     valid;
} vedirect_data_t;

static vedirect_data_t s_data;

/* Parse a single label=value pair */
static void parse_field(const char *label, const char *value)
{
    if (strcmp(label, "V") == 0) {
        s_data.battery_voltage_v = atof(value) / 1000.0f;
    } else if (strcmp(label, "I") == 0) {
        s_data.battery_current_a = atof(value) / 1000.0f;
    } else if (strcmp(label, "VPV") == 0) {
        s_data.panel_voltage_v = atof(value) / 1000.0f;
    } else if (strcmp(label, "PPV") == 0) {
        s_data.panel_power_w = (uint16_t)atoi(value);
    } else if (strcmp(label, "CS") == 0) {
        s_data.charge_state = (uint8_t)atoi(value);
    } else if (strcmp(label, "ERR") == 0) {
        s_data.error_code = (uint8_t)atoi(value);
    } else if (strcmp(label, "H20") == 0) {
        s_data.yield_today_wh = (uint32_t)(atoi(value) * 10);
    }
}

/* Validate VE.Direct checksum: sum of all bytes in frame must = 0 (mod 256) */
static bool validate_checksum(const char *frame, int len)
{
    uint8_t sum = 0;
    for (int i = 0; i < len; i++) {
        sum += (uint8_t)frame[i];
    }
    return (sum == 0);
}

/* Parse a complete TEXT frame (multiple lines terminated by Checksum) */
static bool parse_frame(const char *frame)
{
    memset(&s_data, 0, sizeof(s_data));
    const char *p = frame;

    while (*p) {
        /* Find tab separator */
        const char *tab = strchr(p, '\t');
        if (!tab) break;

        /* Extract label */
        char label[16] = {};
        int llen = (int)(tab - p);
        if (llen > 15) llen = 15;
        memcpy(label, p, llen);

        /* Find end of line */
        const char *eol = strchr(tab + 1, '\r');
        if (!eol) eol = strchr(tab + 1, '\n');
        if (!eol) eol = tab + 1 + strlen(tab + 1);

        /* Extract value */
        char value[32] = {};
        int vlen = (int)(eol - tab - 1);
        if (vlen > 31) vlen = 31;
        memcpy(value, tab + 1, vlen);

        if (strcmp(label, "Checksum") != 0) {
            parse_field(label, value);
        }

        /* Skip to next line */
        p = eol;
        while (*p == '\r' || *p == '\n') p++;
    }

    s_data.valid = true;
    return true;
}

void test_checksum_valid(void)
{
    /* Real VE.Direct frame: sum of all bytes = 0 mod 256 */
    /* Simplified: "A\t1\r\nChecksum\t\xfe" → 'A'+'\t'+'1'+'\r'+'\n'+'C'...'m'+'\t'+0xFE */
    /* For simplicity, test the checksum logic directly */
    uint8_t frame[] = { 0x41, 0x09, 0x31, 0x0D, 0x0A }; /* "A\t1\r\n" */
    uint8_t sum = 0;
    for (int i = 0; i < 5; i++) sum += frame[i];
    uint8_t checksum_byte = (uint8_t)(0 - sum); /* byte that makes sum = 0 */
    uint8_t full[6];
    memcpy(full, frame, 5);
    full[5] = checksum_byte;
    TEST_ASSERT_TRUE(validate_checksum((const char *)full, 6));
}

void test_checksum_invalid(void)
{
    uint8_t frame[] = { 0x41, 0x09, 0x31, 0x0D, 0x0A, 0x00 };
    TEST_ASSERT_FALSE(validate_checksum((const char *)frame, 6));
}

### Task 6: ESP-IDF memory budget script

**Files:**
- Create: `scripts/ci/esp-idf-memory-budget.sh`

Parses `idf.py size` output and fails if app exceeds partition size threshold.

- [ ] **Step 1: Create script**

Create `scripts/ci/esp-idf-memory-budget.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail

# ESP-IDF memory budget gate
# Usage: esp-idf-memory-budget.sh [--flash-max 85]
#
# Checks binary size vs OTA partition (2MB = 0x200000)

FLASH_MAX=${1:-85}
PARTITION_SIZE=$((0x200000))  # 2MB OTA partition

cd "$(dirname "$0")/../../firmware-idf"

echo "=== ESP-IDF Memory Budget Check ==="

# Get binary size
BIN="build/kxkm-bmu.bin"
if [ ! -f "$BIN" ]; then
    echo "ERROR: $BIN not found. Run 'idf.py build' first."
    exit 1
fi

BIN_SIZE=$(stat -f%z "$BIN" 2>/dev/null || stat -c%s "$BIN" 2>/dev/null)
PERCENT=$(( BIN_SIZE * 100 / PARTITION_SIZE ))

echo "Binary: $BIN_SIZE bytes ($PERCENT% of ${PARTITION_SIZE} byte partition)"
echo "Threshold: ${FLASH_MAX}%"

if [ "$PERCENT" -gt "$FLASH_MAX" ]; then
    echo "FAIL: Flash usage ${PERCENT}% exceeds max ${FLASH_MAX}%"
    exit 1
fi

echo "PASS: Flash usage ${PERCENT}% <= ${FLASH_MAX}%"

# RAM check via idf.py size (if available)
if command -v idf.py &>/dev/null; then
    echo ""
    idf.py size 2>/dev/null | grep -E "Total|Used|Available|DRAM" || true
fi

echo "=== Memory budget OK ==="
```

name: ESP-IDF CI

on:
  push:

    runs-on: ubuntu-latest

      - uses: actions/checkout@v4

      - name: Upload binary
        uses: actions/upload-artifact@v4

    needs: build
    runs-on: ubuntu-latest

        with:
          name: kxkm-bmu-firmware
          path: firmware-idf/build/
      - name: Check binary size
        run: |
          BIN="firmware-idf/build/kxkm-bmu.bin"
          PARTITION_SIZE=$((0x200000))
          BIN_SIZE=$(stat -c%s "$BIN")
          PERCENT=$(( BIN_SIZE * 100 / PARTITION_SIZE ))
          echo "Binary: $BIN_SIZE bytes ($PERCENT% of 2MB partition)"
          if [ "$PERCENT" -gt 85 ]; then
            echo "::error::Flash usage ${PERCENT}% exceeds 85% threshold"
            exit 1
          fi
          echo "✅ Flash: ${PERCENT}% (max 85%)"
